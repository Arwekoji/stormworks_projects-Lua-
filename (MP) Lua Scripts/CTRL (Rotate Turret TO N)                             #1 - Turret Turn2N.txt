function onTick()
	CameraAngle = input.getNumber(1)	
	TurretAngle = input.getNumber(2)	
	Fault = input.getNumber(3)	
	MaxTurretSpeed = input.getNumber(4)
	MinAngle = input.getNumber(5)
	MaxAngle = input.getNumber(6)
	TurretSpeed = 0
	
	if math.max(MinAngle, CameraAngle) < TurretAngle-Fault then
		TurretSpeed = -1*MaxTurretSpeed
	end
	if math.min(MaxAngle, CameraAngle) > TurretAngle+Fault then
		TurretSpeed = MaxTurretSpeed
	end
	
	output.setNumber(1, TurretSpeed)	
end
